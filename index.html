async function toggleConnect() {

  if (connected) {

    disconnect();

  } else {

    await connect();

  }

}


async function connect() {

  setStatus('connecting');

  addLog('Connessione a OpenAI Realtime...', 'system');


  try {

    // Mic

    micStream = await navigator.mediaDevices.getUserMedia({

      audio: { echoCancellation: true, noiseSuppression: true, sampleRate: 24000 }

    });


    audioCtx = new AudioContext({ sampleRate: 24000 });


    // Analyser per visualizer

    analyser = audioCtx.createAnalyser();

    analyser.fftSize = 256;

    const src = audioCtx.createMediaStreamSource(micStream);

    src.connect(analyser);


    // AudioWorklet

    await audioCtx.audioWorklet.addModule('audio-processor.js');

    workletNode = new AudioWorkletNode(audioCtx, 'pcm16-processor');

    workletNode.port.onmessage = onMicChunk;

    analyser.connect(workletNode);

    // Non connect a destination (non vogliamo sentire il mic in cuffia)


    // WebSocket

    const model = document.getElementById('initModel').value;

    ws = new WebSocket(

      `wss://api.openai.com/v1/realtime?model=${model}`,

      ['realtime', `openai-insecure-api-key.${API_KEY}`, 'openai-beta.realtime-v1']

    );

    ws.binaryType = 'arraybuffer';


    ws.onopen = () => {

      sendSessionUpdate();

      setStatus('connected');

      connected = true;

      updateConnectBtn();

      addLog('✅ Connesso! Parla pure...', 'system');

      startViz();

    };


    ws.onmessage = onWsMessage;


    ws.onerror = (e) => { toast('❌ Errore WebSocket'); disconnect(); };

    ws.onclose = (e) => {

      if (connected) { addLog('🔌 Connessione chiusa', 'system'); disconnect(); }

    };


  } catch(e) {

    toast('❌ ' + e.message);

    setStatus('error');

    disconnect();

  }

}


function disconnect() {

  connected = false;

  if (ws) { try { ws.close(); } catch(e){} ws = null; }

  if (workletNode) { workletNode.port.postMessage({cmd:'stop'}); workletNode = null; }

  if (micStream) { micStream.getTracks().forEach(t => t.stop()); micStream = null; }

  if (audioCtx) { audioCtx.close(); audioCtx = null; }

  if (vizFrame) { cancelAnimationFrame(vizFrame); vizFrame = null; }

  analyser = null;

  setStatus('disconnected');

  updateConnectBtn();

}

/* ═══════════════════════════════════════════════

   AUDIO IN (mic → ws)

═══════════════════════════════════════════════ */

function onMicChunk(e) {

  if (!ws || ws.readyState !== WebSocket.OPEN || muted) return;

  const pcm = new Int16Array(e.data);

  const b64 = btoa(String.fromCharCode(...new Uint8Array(pcm.buffer)));

  ws.send(JSON.stringify({ type: 'input_audio_buffer.append', audio: b64 }));

}


/* ═══════════════════════════════════════════════

   AUDIO OUT (ws → speaker)

═══════════════════════════════════════════════ */

let audioQueue = [];

let isPlaying = false;


function onWsMessage(e) {

  let msg;

  try { msg = JSON.parse(e.data); } catch { return; }


  switch (msg.type) {


    case 'response.audio.delta':

      if (msg.delta) {

        const bytes = Uint8Array.from(atob(msg.delta), c => c.charCodeAt(0));

        const pcm = new Int16Array(bytes.buffer);

        audioQueue.push(pcm);

        if (!isPlaying) drainAudio();

      }

      break;


    case 'response.audio_transcript.delta':

      if (msg.delta) appendRobotText(msg.delta);

      break;


    case 'response.audio_transcript.done':

      finalizeRobotText();

      break;


    case 'conversation.item.input_audio_transcription.completed':

      if (msg.transcript) addLog(msg.transcript, 'user');

      break;


    case 'response.done':

      const usage = msg.response?.usage;

      if (usage) updateTokens(usage);

      break;


    case 'input_audio_buffer.speech_started':

      setStatus('listening');

      resetSilenceTimer();

      break;


    case 'input_audio_buffer.speech_stopped':

      setStatus('connected');

      startSilenceTimer();

      break;


    case 'response.audio.done':

      setStatus('connected');

      break;


    case 'error':

      toast('⚠️ ' + (msg.error?.message || 'Errore API'));

      break;

  }

}


// PCM16 → Web Audio playback

async function drainAudio() {

  if (!audioCtx || audioQueue.length === 0) { isPlaying = false; return; }

  isPlaying = true;

  setStatus('talking');


  const pcm = audioQueue.shift();

  const float = new Float32Array(pcm.length);

  for (let i = 0; i < pcm.length; i++) float[i] = pcm[i] / (pcm[i] < 0 ? 0x8000 : 0x7FFF);


  const buf = audioCtx.createBuffer(1, float.length, 24000);

  buf.copyToChannel(float, 0);


  // REC

  if (isRecording) recChunks.push(pcm);


  const src = audioCtx.createBufferSource();

  src.buffer = buf;

  src.connect(audioCtx.destination);

  src.onended = () => drainAudio();

  src.start();

}

function startViz() {

  const W = () => vizCanvas.offsetWidth;

  const H = () => vizCanvas.offsetHeight;


  function draw() {

    vizFrame = requestAnimationFrame(draw);

    if (!analyser) return;

    const data = new Uint8Array(analyser.frequencyBinCount);

    analyser.getByteFrequencyData(data);


    vizCtx.fillStyle = '#0f172a';

    vizCtx.fillRect(0, 0, W(), H());


    const barW = W() / data.length * 2.5;

    let x = 0;

    for (let i = 0; i < data.length; i++) {

      const h = (data[i] / 255) * H();

      const g = vizCtx.createLinearGradient(0, H() - h, 0, H());

      g.addColorStop(0, '#10a37f');

      g.addColorStop(1, '#0d7f6322');

      vizCtx.fillStyle = g;

      vizCtx.fillRect(x, H() - h, barW - 1, h);

      x += barW;

    }

  }

  draw();

}

function toggleRec() {

  isRecording = !isRecording;

  const btn = document.getElementById('btnRec');

  if (isRecording) {

    recChunks = [];

  } else {

    btn.style.color = '';

    if (recChunks.length > 0) downloadWav();

    toast('💾 WAV salvato');

  }

}


function downloadWav() {

  // Unisci tutti i chunk PCM16 con 1s di silenzio tra ognuno

  const silence = new Int16Array(recSampleRate); // 1s silenzio

  const parts = [];

  recChunks.forEach((c, i) => {

    parts.push(c);

    if (i < recChunks.length - 1) parts.push(silence);

  });


  const total = parts.reduce((a, c) => a + c.length, 0);

  const pcm   = new Int16Array(total);

  let off = 0;

  parts.forEach(p => { pcm.set(p, off); off += p.length; });


  const wav = encodeWav(pcm, recSampleRate);

  const url = URL.createObjectURL(new Blob([wav], { type: 'audio/wav' }));

  const a   = document.createElement('a');

  a.href    = url;

  a.download = `robot-rec-${Date.now()}.wav`;

  a.click();

  URL.revokeObjectURL(url);

}


function encodeWav(pcm, sr) {

  const len    = pcm.length;

  const buf    = new ArrayBuffer(44 + len * 2);

  const view   = new DataView(buf);

  const write  = (off, str) => [...str].forEach((c, i) => view.setUint8(off + i, c.charCodeAt(0)));

  write(0, 'RIFF'); view.setUint32(4, 36 + len * 2, true);

  write(8, 'WAVE'); write(12, 'fmt ');

  view.setUint32(16, 16, true); view.setUint16(20, 1, true);

  view.setUint16(22, 1, true); view.setUint32(24, sr, true);

  view.setUint32(28, sr * 2, true); view.setUint16(32, 2, true);

  view.setUint16(34, 16, true); write(36, 'data');

  view.setUint32(40, len * 2, true);

  for (let i = 0; i < len; i++) view.setInt16(44 + i * 2, pcm[i], true);

  return buf;

}
